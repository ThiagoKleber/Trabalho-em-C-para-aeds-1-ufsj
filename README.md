int apagar2(Cadastrofps y[999],int nfps){
	int r;
	printf("!!!!APAGAR PERSONAGEM!!!!\n");
	printf("Digite o numero do personagem que deseja apagar: ");
	scanf("%d", &r);
	r=r-1;
	if(r >= nfps){
		printf("Digite um numero valido.\n");
		return 0;
	}
	for(int i = r; i < nfps; i++){
		y[i]=y[i+1];
		
	}
	printf("Personagem apagado\n");
	  return 1;
}
